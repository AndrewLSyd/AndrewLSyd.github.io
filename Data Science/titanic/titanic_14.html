class imputer(TransformerMixin):
    def fit(self, X, y=None):
        """ does nothing. we are only transforming"""
        return self
    
    def transform(self ,X, y=None):
        median = SimpleImputer(strategy='median')
        X.loc[:, ["Age"]] = median.fit_transform(X.loc[:, ["Age"]])                                              
        X.loc[:, ["Fare"]] = median.fit_transform(X.loc[:, ["Fare"]])

        most_frequent = SimpleImputer(strategy='most_frequent')
        X.loc[:, ["Cabin"]] = most_frequent.fit_transform(X.loc[:, ["Cabin"]])
        X.loc[:, ["Embarked"]] = most_frequent.fit_transform(X.loc[:, ["Embarked"]])
        
        return X

class feature_engineer(TransformerMixin):
    def __init__(self):
        self._title_list = ['Mrs', 'Mr', 'Master', 'Miss', 'Major', 'Rev', 'Dr', 'Ms', 'Mlle','Col', 'Capt', 'Mme', 'Countess', 'Don', 'Jonkheer']        
        self._cabin_list = ['A', 'B', 'C', 'D', 'E', 'F', 'T', 'G', 'Unknown']

    def substrings_in_string(self, big_string, substrings):
        """ helper function for title"""
        for substring in substrings:
            if str.find(big_string, substring) != -1:
                return substring
        print(big_string)
        return np.nan   
    
    
#     Index(['Age', 'SibSp', 'Parch', 'Fare', 'family_size', 'fare_per_person',
#        'under_16', 'PClass_2', 'PClass_3', 'Sex_male', 'Embarked_Q',
#        'Embarked_S', 'Title_Don', 'Title_Dr', 'Title_Master', 'Title_Miss',
#        'Title_Mr', 'Title_Mrs', 'Title_Ms', 'Title_Rev', 'Deck_B', 'Deck_C',
#        'Deck_D', 'Deck_E', 'Deck_F', 'Deck_G'],
#       dtype='object')
    
    def replace_titles(self, x):
        title=x['Title']
        if title in ['Don', 'Major', 'Capt', 'Jonkheer', 'Col']:
            return 'Mr'
        elif title in ['Countess', 'Mme']:
            return 'Mrs'
        elif title in ['Mlle', 'Ms']:
            return 'Miss'
        else:
            return title

    def fit(self, X, y=None):
        """ does nothing. we are only transforming"""
        return self
    
    def transform(self ,X, y=None):
        # family size
        X.loc[:,'family_size'] = X.loc[:, 'SibSp'] + X.loc[:, 'Parch']
        # fare_per_person
        X.loc[:,'fare_per_person'] = X.loc[:, 'Fare'] / (X.loc[:, 'family_size'] + 1)   
        # title
        # code from: https://triangleinequality.wordpress.com/2013/09/08/basic-feature-engineering-with-the-titanic-data/
        X.loc[:,'Title'] = X.loc[:, 'Name'].map(lambda x: self.substrings_in_string(x, self._title_list))        
        
        X.loc[:,'Title'] = X.apply(self.replace_titles, axis=1)       
#      
        
        # ticket to deck        
        X.loc[:,'Deck'] = X.loc[:,'Cabin'].astype(str).map(lambda x: self.substrings_in_string(x, self._cabin_list))        
        # age grouping base on one way
#         X.loc[:, 'age_0_16'] = (X.loc[:, 'Age'] < 16) + 0
        X.loc[:, 'age_17_50'] = (X.loc[:, 'Age'] > 16) & (X.loc[:, 'Age'] <= 50) + 0
        X.loc[:, 'age_51_'] = (X.loc[:, 'Age'] >50) + 0
        # one hot encoding
        ohe = pd.get_dummies(X.loc[:,'Pclass'], prefix='PClass', drop_first=True)
        X = pd.concat([X, ohe], sort=False, axis='columns')     
        ohe = pd.get_dummies(X.loc[:,'Sex'], prefix='Sex', drop_first=True)
        X = pd.concat([X, ohe], sort=False, axis='columns')
        ohe = pd.get_dummies(X.loc[:,'Embarked'], prefix='Embarked', drop_first=True)
        X = pd.concat([X, ohe], sort=False, axis='columns')
        ohe = pd.get_dummies(X.loc[:,'Title'], prefix='Title', drop_first=True)
        X = pd.concat([X, ohe], sort=False, axis='columns')
        ohe = pd.get_dummies(X.loc[:,'Deck'], prefix='Deck', drop_first=True)        
        X = pd.concat([X, ohe], sort=False, axis='columns')        
        
        # drop useless ID vars as well as one hot encoded variables
        X = X.drop(['Name', 'Pclass', 'Sex', 'Embarked', 'Ticket', 'Cabin', 'Title', 'Deck'], axis='columns')        
        X = X.drop(['Fare'], axis='columns')        
        
        return X

# Import KFold
from sklearn.model_selection import StratifiedKFold
import statistics

def cross_validate(model, X, y, X_valid, y_valid, folds=10, seed=123, verbose=False, return_list=False):
    """ inputs:
            model - a model object adhereing to scikilearns model class structure
            X - features
            y - response
        fits the model the X with y as a response with cross-validation.
        returns the CV accuracy of the model
    """
    kf = StratifiedKFold(n_splits=folds, shuffle=True, random_state=seed)
    fold = 0
    results = []
    
    for train_index, test_index in kf.split(X, y):    
        cv_train, cv_test = X.iloc[train_index], X.iloc[test_index]
        cv_train_target, cv_test_target = y.iloc[train_index], y.iloc[test_index]
               
        model.fit(X=cv_train, y=cv_train_target)
        pred = model.predict(cv_test)
        accuracy = accuracy_score(cv_test_target, pred)
        
        if verbose:
            print('Fold: {}'.format(fold))
            print('CV train shape: {}'.format(cv_train.shape))
            print('Fold', fold, 'accuracy', accuracy)

        results.append(accuracy)       
        fold += 1

    if return_list:
        return results
    
    cv_accuracy = sum(results) / len(results)
    cv_std = statistics.stdev(results)
    
    if verbose:
        print("CV accuracy is", cv_accuracy, "with a standard dev. of", cv_std)
    
    model.fit(X, y)    
    
    return [cv_accuracy, cv_std, model.score(X_valid, y_valid)]

def grid_search_w_valid(model, param_dict):
    """ perform grid search across multiple feature lists"""
    gs_list = []

    features_list_list = [features, features_log_reg, features_k_best]

    for feature_list in features_list_list:
        gs = GridSearchCV(model, param_dict, verbose=1, scoring='accuracy', n_jobs=8, refit = True, cv=5)
        gs.fit(train[feature_list],train.Survived)

        gs_list.append(gs)
    
    return gs_list

Model blending and Final Submission¶

We now combine our 3 best models and use a majority vote as the final predicted class.

In [265]:
def score_majority_vote(X, y=None):
    """ score using majority vote"""
    # all models use the lasso regressions feature list except NN MLP and Ada Boost
    pred_log_reg = log_reg_gs[1].predict(X[features_log_reg])
    pred_xgb = xgb_gs[1].predict(X[features_log_reg])
    pred_nnmlp = nnmlp_gs[0].predict(X[features])
    pred_rf = rf_gs[1].predict(X[features_log_reg])
    pred_ada = ada_gs[0].predict(X[features])

    if y is None:
        df = pd.DataFrame({
            'pred_log_reg':pred_log_reg,
            'pred_xgb':pred_xgb,
            'pred_nnmlp':pred_nnmlp,
            'pred_rf':pred_rf,
            'pred_ada':pred_ada
        })
    else:
        df = pd.DataFrame({
            'pred_log_reg':pred_log_reg,
            'pred_xgb':pred_xgb,
            'pred_nnmlp':pred_nnmlp,
            'pred_rf':pred_rf,
            'pred_ada':pred_ada,
            'Survived':y})
        
    df['majority_pred'] = df['pred_log_reg'] + df['pred_xgb'] + df['pred_nnmlp'] + df['pred_rf'] + df['pred_ada']

    df['majority_pred'] = df['majority_pred'] / 5
    
    df['majority_pred'] = df['majority_pred'].round(0)
    return df
